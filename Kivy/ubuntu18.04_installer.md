# Installer

```python
#!/usr/bin/python
import os

os.system("sudo add-apt-repository ppa:kivy-team/kivy")
os.system("sudo apt update -y")
os.system("sudo apt install python-kivy")
os.system("sudo apt install python3-kivy")
os.system("sudo apt install kivy-examples")
os.system("apt update -y")
print('Done ;)')
```
