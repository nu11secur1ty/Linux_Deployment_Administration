# Installer

```python
#!/usr/bin/python
import os

os.system("sudo add-apt-repository ppa:kivy-team/kivy")
os.system("sudo apt update -y")
os.system("sudo apt install python-kivy -y")
os.system("sudo apt install python3-kivy -y")
os.system("sudo apt install kivy-examples -y")
os.system("apt update -y")
os.system("pip install kivy")
print('Done ;)')
```
